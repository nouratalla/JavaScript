const myObject = {
    name:'nour',
    age:24,
    favSkill: 'js',
    [Symbol.iterator] () {
        let step = 0;
        let properties = Object.keys(this);
        return {
            next(){
                return {
                    value:properties[step],
                    done: step ++ === properties.length
                }
            }
        }
    }
}

for ( let prop of myObject){
    console.log(prop);
}
